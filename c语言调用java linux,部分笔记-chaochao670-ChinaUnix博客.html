c语言调用java linux,部分笔记 2013-03-21 13:34:31

分类： C/C++

初步学习了c语言调用java，原理的介绍很多文章有，这里仅记录基本使用方法，c语言调用编译出来的class文件

linux下安装过java后，java文件夹中自带的头文件和动态库可以使c语言调用java

编写hello.java文件

public class hello
{
        public void hello()
        {   
                System.out.println("hello() constructor run ...");
        }   
        public void print()
        {   
                System.out.println("hello.print() run");
        }   
        public static void StaticFunction()
        {   
                System.out.println("--- static void StaticFunction() run");
        }   
        public static int add(int a,int b)
        {   
                int c=a+b;
                System.out.println("--- static int add(int,int) run ---");
                System.out.println( a + "+"+ b + "=" + c );
                return a + b;
        }   
        public static int showString(String str)
        {   
                System.out.println("--- static int showString(String) run ---");
                System.out.println("static showString() receive -->"+ str + " <--");
                return 0;
        }   
}
javac hello.java编译该文件，生成hello.class的类文件

编译cCallJava.c的c源码，和hello.class在同一文件夹下

#include <jni.h>  
#include <stdlib.h>  
#include <stdio.h> 
int main(int argc,char* argv[])  
{  
        JavaVMOption options[1];  
        JNIEnv *env;  //c语言编写一个java虚拟机跑hello.class文件
        JavaVM *jvm;  //找classID和调用class方法来自这里
        JavaVMInitArgs vm_args;  
        long status;  
        jclass cls;  
        jmethodID mid;  
        jint square;  
        jobjectArray applicationArgs;

        options[0].optionString = "-Djava.class.path=.";  //path=.表明类路径（如hello.class）在当前路径，可以接收路径参数
        vm_args.version = 0x00010002;  //也可以写成JNI_VERSION_1_2;的宏，指明所用的虚拟机版本
        vm_args.nOptions = 1;  
        vm_args.options = options;  
        status = JNI_CreateJavaVM(&jvm, (void**)&env, &vm_args);  

        if (status != JNI_ERR)  
        {   
                printf("create JVM successfully\n");//创建java虚拟机成功
                cls = (*env)->FindClass(env,"hello");  //在当前路径.中寻找hello.class
                if(cls !=0)  
                {   
                        printf("find class successfully\n");//成功找到hello.class
                        mid = (*env)-& gt;GetStaticMethodID(env, cls, "StaticFunction", "()V");   //找名为 StaticFunction，接收参数为()返回类型为V(oid)的静态函数ID,这些方法可用javap -s hello来查看出所有可调用方法 以及参数列表
                        if(mid !=0)  
                        {    
                                printf("get static method successfully\n");//找到
                                (*env)->CallStaticVoidMethod(env, cls, mid, applicationArgs);// 调用，applicationArgs是该函数如果需要参数，用来
传参数
                        }   

                }   
                (*jvm)->DestroyJavaVM(jvm);   //摧毁虚拟机
                return 0;  
        }   
        else  
                return -1;  

}

使用动态库时，linux需指定动态库路径的环境变量,即寻找动态库的附加路径

export LD_LIBRARY_PATH=/home/chao/java/jdk1.6.0_27/jre/lib/i386/client


编译时候需指定头文件路径和连接的动态库名

gcc -o jvm cCallJava.c -I/home/chao/java/jdk1.6.0_27/include/linux -I/home/chao/java/jdk1.6.0_27/include -L/home/chao/java/jdk1.6.0_27/jre/lib/i386/client -ljvm

-I告诉编译器，在后面附加路径里面找头文件，如头文件jni.h,-L告诉编译器到后面附加路径里找动态库*.so文件，-ljvm指明需要连接的库名为libjvm.so,

编译完成后运行./jvm可以看到c调hello.class

#include <jni.h>
#include <stdlib.h>
#include <stdio.h>
int main(int argc,char* argv[])
{
        JavaVMOption options[1];
        JNIEnv *env;
        JavaVM *jvm;
        JavaVMInitArgs vm_args;
        long status;
        jclass cls;
        jmethodID mid;
        jint square;
        jobjectArray applicationArgs;
        jstring appArg;

        options[0].optionString = "-Djava.class.path=.";
        vm_args.version = 0x00010002;
        vm_args.nOptions = 1;
        vm_args.options = options;
        status = JNI_CreateJavaVM(&jvm, (void**)&env, &vm_args);

        if (status != JNI_ERR)
        {
                printf("create JVM successfully\n");
                cls = (*env)->FindClass(env,"hello");
                if(cls !=0)
                {
                        printf("find class successfully\n");
// ----------------------- 1.调用静态空方法 -------------------------
                        mid = (*env)->GetStaticMethodID(env, cls, "StaticFunction", "()V");
                        if(mid !=0)
                        {
                                printf("[1].get static void method successfully\n");
                                (*env)->CallStaticVoidMethod(env, cls, mid, applicationArgs);
                        }
// ----------------------- 1.调用静态空方法 -------------------------

// ----------------------- 2.调用静态接收整型参数方法 -------------------------
                        mid = (*env)->GetStaticMethodID(env, cls, "add", "(II)I");
                        if(mid !=0)
                        {
                                printf("[2].get static int add method successfully\n");
                                square= (*env)->CallStaticIntMethod(env, cls, mid, 3,4);
                                printf("c call int add(int,int) return value = %d \n",square);
                        }
// ----------------------- 2.调用静态接收整型参数方法 -------------------------

// ----------------------- 3.调用静态接收字符串方法 -------------------------
                        mid = (*env)->GetStaticMethodID(env, cls, "showString", "(Ljava/lang/String;)I");
                        if(mid !=0)
                        {
                                printf("[3].get static int method successfully\n");
                                if (argc>1)
                                {
                                        appArg = (*env)->NewStringUTF(env, argv[1]);
                                        square = (*env)->CallStaticIntMethod(env, cls, mid, appArg);
                                        printf("c call ShowString return value = %d \n",square);
                                }
                        }
// ----------------------- 3.调用静态接收字符串方法 -------------------------

// ----------------------- 4.调用类对象方法 -------------------------
                        jmethodID ctor= (*env)->GetMethodID(env,cls,"<init>","()V"); //查找构造函数
                        if(ctor !=0)
                        {
                                printf("[4].get hello() constructor method successfully\n");
                                jobject jobj;
                                jobj=(*env)->NewObject(env,cls,ctor);//new一个类对象出来 
                                mid=(*env)->GetMethodID( env,cls, "print","()V");
                                {
                                        printf("get class.method successfully\n");
                                        (*env)->CallVoidMethod(env,jobj,mid,applicationArgs);//对对象调用方法
                                }
                        }
// ----------------------- 4.调用类对象方法 -------------------------

                }
                (*jvm)->DestroyJavaVM(jvm);
                return 0;
        }
        else
                return -1;

}

3.上面是向java传参数使用方法，下面是接收java返回 结构体和String参数 的方法(返回int的见方法2中square)

编写hello.java

public class hello
{
    public String name;
    public int serial;

    public static hello get()
    {
        System.out.println("get() run");

        hello d1=new hello();
        d1.name="disk name";
        d1.serial=12;
        return d1;
    }
    public static String getS()
    {
        System.out.println("getS run");

        String s1="I am return string";
        return s1;
    }
}

javac hello.java,生成hello.class
现用c调用hello.class,获取String和hello对象的返回值

编写c文件

#include <jni.h>  
#include <stdlib.h>  
#include <stdio.h> 
int main(int argc,char* argv[])  
{  
    JavaVMOption options[1];  
    JNIEnv *env;  
    JavaVM *jvm;  
    JavaVMInitArgs vm_args;  
    long status;  
    jclass cls;  
    jmethodID mid;  
    jobjectArray applicationArgs;
    jstring appArg;
    jstring msg;
    char* test;

    options[0].optionString = "-Djava.class.path=/tmp";  
    vm_args.version = 0x00010002;  
    vm_args.nOptions = 1;  
    vm_args.options = options;  
    status = JNI_CreateJavaVM(&jvm, (void**)&env, &vm_args);  

    if (status != JNI_ERR)  
    {  
        printf("create JVM successfully\n");
        cls = (*env)->FindClass(env,"hello");  
        if(cls != 0)
        {
            printf("find classs successsfully\n");

            jmethodID ctor= (*env)->GetMethodID(env,cls,"<init>","()V"); //查找构造函数
            if(ctor !=0)
            {
                printf("get constructor successfully\n");
                jmethodID mid=(*env)->GetStaticMethodID( env,cls, "get","()Lhello;");
                if(mid!=0)
                {
                    printf("get function get successfully\n");

                    jobject jobj;
                    jobj=(*env)->NewObject(env,cls,ctor);//new一个类对象用以接收返回值

                    jobj=(*env)->CallStaticObjectMethod(env, cls, mid, appArg);//调用get方法，接收返回值

                    jfieldID id_number = (*env)->GetFieldID(env,cls,"serial","I");
                    jint number = (*env)->GetIntField(env,jobj, id_number); //获取类对象中的成员变量,serial

                    jfieldID id_name = (*env)->GetFieldID(env,cls,"name","Ljava/lang/String;");
                    msg= (*env)->GetObjectField(env,jobj,id_name);//获取类对象中的成员变量name

                    test = (char*)(*env)->GetStringUTFChars(env, msg, NULL);

                    printf("after %d-%s\n",number,test);
                }

                mid=(*env)->GetStaticMethodID( env,cls, "getS","()Ljava/lang/String;");//调用返回string的方法
                if(mid!=0)
                {   
                    printf("find string function successfully\n");
                    msg = (jstring)(*env)->CallObjectMethod(env, cls, mid);//jstring接收返回的String

                    test = (char*)(*env)->GetStringUTFChars(env, msg, NULL);//jstring转为char*
                    printf("return string >%s<\n",test);
                }   
            }
        }

    }  
}
